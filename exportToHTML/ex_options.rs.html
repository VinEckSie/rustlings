205  fn check_option(optional: Option<Box<i32>>) {
206      match optional {
207          Some(x) => println!("Value: {}", x),
208          None => println!("None"),
209      }

212  //? operator (shorter way as match pattern)
213  // Use ? when…  
214  // You just want to propagate None upwards | You’re in a function that returns Option | You want cleaner, linear code | You want fewer nested blocks
215  // Use match when…
216  // You want to handle None differently | You're doing something more complex | You’re matching on multiple values at once | You need different logic per match arm
217  fn get_value(optional: Option<String>)  -> Option<String> {
218      Some(optional)?
219  }

221  //NPO (Nullable pointer optimization)
222  // Concept                                  What to remember
223  // Option<Box<T>>, Option<&T>               Zero-cost — same size as the pointer
224  // Use Option for optional fields           It’s safe and efficient
225  // #[repr(transparent)] structs             Inherit optimization from inner type
226  // Don't worry about transmute              Unless doing unsafe systems-level work
227  // Safe nullable pointers                   Use Option always — no null bugs
